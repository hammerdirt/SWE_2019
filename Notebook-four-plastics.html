# adding a pcs/m column
regional_data_df['pcs_m'] = regional_data_df['quantity']/regional_data_df['length']
plastic_data_df['pcs_m'] = plastic_data_df['quantity']/plastic_data_df['length']
# grouping the results by location and date and adding up the total for eash group
all_trash = regional_data_df[['location_id','date', 'pcs_m']].groupby(['date', 'location_id']).sum()
just_plastic = plastic_data_df[['location_id', 'date', 'pcs_m']].groupby(['date', 'location_id']).sum()
# resetting index so the columns can be used in a chart
all_trash.reset_index(inplace=True)
just_plastic.reset_index(inplace=True)
